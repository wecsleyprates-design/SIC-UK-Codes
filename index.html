The Global Extraction Model

By implementing a "Prefix-Aware" adapter pattern, we can scale this solution to any country.

integration-service / lib/facts/businessDetails/index.ts
uk_sic_code: [
  {
    source: sources.opencorporates,
    fn: (_, oc: OpenCorporateResponse) => {
      // Logic: Loop through registry strings via pipe separator
      for (const uid of oc.industry_code_uids.split("|")) {
        const [prefix, code] = uid.split("-", 2);
        // Solution: Map 'gb_sic' prefix specifically
        if (prefix === "gb_sic" && /^\d+/.test(code)) {
          return code.padStart(5, "0"); // Normalize to 5-digit UK Standard
        }
      }
    }
  }
]
